176impl Pins {
177    /// Create an instance of `Pins` when you do not have a handle
178    /// to the processor pads
179    ///
180    /// # Safety
181    ///
182    /// Caller must ensure that the pins are not aliased elsewhere in
183    /// the program. This could include
184    ///
185    /// - an existing handle to the `imxrt-iomuxc` pads,
186    /// - another instance of `Pins` that was safely acquired
187    ///   using [`from_pads`](from_pads()).
188    #[inline]
189    pub const unsafe fn new() -> Self {
190        from_pads(crate::iomuxc::Pads::new())
191    }
